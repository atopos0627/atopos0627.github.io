# -*- coding: utf-8 -*- 
import os

def Test1(rootDir): 
    list_dirs = os.walk(rootDir) 
    for root, dirs, files in list_dirs: 
        for d in dirs: 
            print os.path.join(root, d)      
        for f in files: 
            print os.path.join(root, f) 


def Test2(rootDir):

Ubuntu下Android手机通过USB连接PC方法

参考： Ubuntu下Android手机通过USB连接PC方法

  1. 查找手机usb信息（ID） 
    lsusb
  2. 添加udev规则信息
    sudo /etc/udev/rules.d/70-android-usb.rules
添加内容：SUBSYSTEM=="usb",ATTRS{idVendor}=="18d1",ATTRS{idProduct}=="9025",MODE="0666"
or 
添加内容 SUBSYSTEM=="usb", SYSFS("My Phone (USB Debug)")=="18d1",MODE="0666"
sudo chmod a+rx /etc/udev/rules.d/70-android-usb.rules
sudo /etc/init.d/udev restart
