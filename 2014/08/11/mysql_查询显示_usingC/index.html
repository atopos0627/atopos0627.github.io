利用python或者C++将数据存入数据库，利用PHP可以实现B/S的查询系统，也可以利用C#等语言实现本地执行的应用程序进行数据库查询。本篇主要讨论学习利用C#实现mysql数据库的打开，查询和显示。

环境准备

  • VS2010 2012 2013
  • mysql-connect-net

打开数据库

MySql.Data.MySqlClient.MySqlConnection conn;
string myConnectionString;
myConnectionString = “server=127.0.0.1;uid=root;” + “pwd=12345;database=test;”;
try
{
conn = new MySql.Data.MySqlClient.MySqlConnection(myConnectionString);
conn.Open();
}
catch (MySql.Data.MySqlClient.MySqlException ex)
{
MessageBox.Show(ex.Message);
}

使用 mysqlcommand 插入数据到数据库

string connStr = “server=localhost;user=root;database=world;port=3306;password=******;”;
MySqlConnection conn = new MySqlConnection(connStr);
try
{
Console.WriteLine(“Connecting to MySQL…”);
conn.Open();
string sql = “INSERT INTO Country (Name, HeadOfState, Continent) VALUES (‘Disneyland’,’Mickey Mouse’, ‘North America’)”;
MySqlCommand cmd = new MySqlCommand(sql, conn);
cmd.ExecuteNonQuery();
}
catch (Exception ex)
{
Console.WriteLine(ex.ToString());
}
conn.Close();
Console.WriteLine(“Done.”);

使用 mysqlcommand 获取表中的数目

string connStr = “server=localhost;user=root;database=world;port=3306;password=******;”;
MySqlConnection conn = new MySqlConnection(connStr);
try
{
Console.WriteLine(“Connecting to MySQL…”);
conn.Open();
string sql = “SELECT COUNT(*) FROM Country”;
MySqlCommand cmd = new MySqlCommand(sql, conn);
object result = cmd.ExecuteScalar();
if (result != null)
{
int r = Convert.ToInt32(result);
Console.WriteLine(“Number of countries in the World database is: “ + r);
}
}
catch (Exception ex)
{
Console.WriteLine(ex.ToString());
}
conn.Close();
Console.WriteLine(“Done.”);
